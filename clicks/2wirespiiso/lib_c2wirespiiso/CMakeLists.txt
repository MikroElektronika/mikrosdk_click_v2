cmake_minimum_required(VERSION 3.21)
if (${TOOLCHAIN_LANGUAGE} MATCHES "MikroC")
    project(lib_c2wirespiiso LANGUAGES MikroC)
else()
    project(lib_c2wirespiiso LANGUAGES C ASM)
endif()

if (NOT PROJECT_TYPE)
    set(PROJECT_TYPE "mikrosdk" CACHE STRING "" FORCE)
endif()

add_library(lib_c2wirespiiso STATIC
        src/c2wirespiiso.c
        include/c2wirespiiso.h
)
add_library(Click.2WireSPIISO  ALIAS lib_c2wirespiiso)

if (${CORE_NAME} MATCHES "DSPIC")
    if (${MCU_NAME} MATCHES "(^PIC24FV?(0[48]|16)K[ML][1234](.+)$)")
        set(sample_mode_reg_case "SAMPLE_REG_CASE_2")
    else()
        set(sample_mode_reg_case "SAMPLE_REG_CASE_1")
    endif()
else()
    if (${MCU_NAME} MATCHES "(^PIC18F(.+)[KQ]4[23]$)|(^PIC32(.+)$)")
        set(sample_mode_reg_case "SAMPLE_REG_CASE_1")
    else()
        set(sample_mode_reg_case "SAMPLE_REG_CASE_2")
    endif()
endif()

target_compile_definitions(lib_c2wirespiiso PUBLIC ${sample_mode_reg_case})


target_include_directories(lib_c2wirespiiso PUBLIC 
        include
)

find_package(MikroC.Core REQUIRED)
target_link_libraries(lib_c2wirespiiso PUBLIC MikroC.Core)
find_package(MikroSDK.Driver REQUIRED)
target_link_libraries(lib_c2wirespiiso PUBLIC MikroSDK.Driver)
