cmake_minimum_required(VERSION 3.21)
if (${TOOLCHAIN_LANGUAGE} MATCHES "MikroC")
    project(lib_charger30 LANGUAGES MikroC)
else()
    project(lib_charger30 LANGUAGES C ASM)
endif()

if (NOT PROJECT_TYPE)
    set(PROJECT_TYPE "mikrosdk" CACHE STRING "" FORCE)
endif()

add_library(lib_charger30 STATIC
        src/charger30.c
        include/charger30.h
)
add_library(Click.Charger30  ALIAS lib_charger30)



target_include_directories(lib_charger30 PUBLIC 
        include
)

find_package(MikroC.Core REQUIRED)
target_link_libraries(lib_charger30 PUBLIC MikroC.Core)
find_package(MikroSDK.Driver REQUIRED)
target_link_libraries(lib_charger30 PUBLIC MikroSDK.Driver)
