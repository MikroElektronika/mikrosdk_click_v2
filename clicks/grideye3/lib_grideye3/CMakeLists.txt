cmake_minimum_required(VERSION 3.21)
if (${TOOLCHAIN_LANGUAGE} MATCHES "MikroC")
    project(lib_grideye3 LANGUAGES MikroC)
else()
    project(lib_grideye3 LANGUAGES C ASM)
endif()

if (NOT PROJECT_TYPE)
    set(PROJECT_TYPE "mikrosdk" CACHE STRING "" FORCE)
endif()

add_library(lib_grideye3 STATIC
        src/grideye3.c
        include/grideye3.h
)
add_library(Click.GridEYE3  ALIAS lib_grideye3)



target_include_directories(lib_grideye3 PUBLIC 
        include
)

find_package(MikroC.Core REQUIRED)
target_link_libraries(lib_grideye3 PUBLIC MikroC.Core)
find_package(MikroSDK.Driver REQUIRED)
target_link_libraries(lib_grideye3 PUBLIC MikroSDK.Driver)
