cmake_minimum_required(VERSION 3.21)
if (${TOOLCHAIN_LANGUAGE} MATCHES "MikroC")
    project(lib_smartbuck8 LANGUAGES MikroC)
else()
    project(lib_smartbuck8 LANGUAGES C ASM)
endif()

if (NOT PROJECT_TYPE)
    set(PROJECT_TYPE "mikrosdk" CACHE STRING "" FORCE)
endif()

add_library(lib_smartbuck8 STATIC
        src/smartbuck8.c
        include/smartbuck8.h
)
add_library(Click.SmartBuck8  ALIAS lib_smartbuck8)



target_include_directories(lib_smartbuck8 PUBLIC 
        include
)

find_package(MikroC.Core REQUIRED)
target_link_libraries(lib_smartbuck8 PUBLIC MikroC.Core)
find_package(MikroSDK.Driver REQUIRED)
target_link_libraries(lib_smartbuck8 PUBLIC MikroSDK.Driver)
