cmake_minimum_required(VERSION 3.21)
if (${TOOLCHAIN_LANGUAGE} MATCHES "MikroC")
    project(lib_stepper28 LANGUAGES MikroC)
else()
    project(lib_stepper28 LANGUAGES C ASM)
endif()

if (NOT PROJECT_TYPE)
    set(PROJECT_TYPE "mikrosdk" CACHE STRING "" FORCE)
endif()

add_library(lib_stepper28 STATIC
        src/stepper28.c
        include/stepper28.h
)
add_library(Click.Stepper28  ALIAS lib_stepper28)



target_include_directories(lib_stepper28 PUBLIC 
        include
)

find_package(MikroC.Core REQUIRED)
target_link_libraries(lib_stepper28 PUBLIC MikroC.Core)
find_package(MikroSDK.Driver REQUIRED)
target_link_libraries(lib_stepper28 PUBLIC MikroSDK.Driver)
